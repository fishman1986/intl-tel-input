Example: Different countryMode

Set countryMode to use intlTelInput in different ways

Markup

<table>
    <tr>
        <td>Full:</td>
        <td>
            <input name="phone" id="phone-full" type="tel"> <br/>
        </td>
    </tr>
    <tr>
        <td>Single:</td>
        <td>
            <input name="phone" id="phone-single" type="tel"> <br/>
        </td>
    </tr>
    <tr>
        <td>None:</td>
        <td>
            <input name="phone" id="phone-none" type="tel"> <br/>
        </td>
    </tr>
</table>

Code

var input1 = $("#phone-full"),
    input2 = $("#phone-single"),
    input3 = $("#phone-none");

input1.intlTelInput({
    utilsScript: "../../lib/libphonenumber/build/utils.js" // just for formatting/placeholders etc
});
input2.intlTelInput({
    countryMode:'single'
});
input3.intlTelInput({
    countryMode:'none'
});
